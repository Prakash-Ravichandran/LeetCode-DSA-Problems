class Solution {
     public static String sortString(String inputString)
    {
        // Converting input string to character array
        char tempArray[] = inputString.toCharArray();
 
        // Sorting temp array using
        Arrays.sort(tempArray);
 
        // Returning new sorted string
        return new String(tempArray);
    }    
    
     public static boolean canConstruct(String ransomNote, String magazine) {
        boolean isCanConstruct = false;
        char[] strArr = new char[magazine.length()];
        String sortedMagazine = sortString(magazine);
        System.out.println("sorted Magazine="+sortedMagazine);
  
        for(int i=0; i <  sortedMagazine.length(); i++){
             strArr[i] =  sortedMagazine.charAt(i);
        }
        String formedString = "";  
        for(int i=0 ; i <  sortedMagazine.length(); i++){
             formedString = formedString + strArr[i];
             System.out.println("formedString is="+formedString);
             if(formedString.equals(ransomNote)){
                 return isCanConstruct = true;
             }
             else{
                 continue;
             }
        }
        return isCanConstruct;
    }
}
